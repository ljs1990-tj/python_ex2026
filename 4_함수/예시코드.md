# [파이썬 기초: 함수(Function) 실습 예제]

# ---------------------------------------------------------
# 1. 가장 기본적인 함수 (매개변수 X, 리턴 X)
# ---------------------------------------------------------
print("--- [1. 기본 함수] ---")

def say_hello():
    # 함수 내부의 실행문은 반드시 들여쓰기를 해야 합니다.
    print("안녕하세요! 함수를 호출하셨군요.")
    print("반갑습니다.")

# 함수를 정의만 하면 실행되지 않습니다. 이름을 불러줘야(호출) 합니다.
say_hello() 
print()


# ---------------------------------------------------------
# 2. 매개변수가 있는 함수 (데이터 전달받기)
# ---------------------------------------------------------
print("--- [2. 매개변수 활용] ---")

def profile(name, age):
    print("이름:", name, "/ 나이:", age)

# 함수를 호출할 때 괄호 안에 값을 넣어 전달합니다.
profile("짱구", 5)
profile("철수", 7)
print()


# ---------------------------------------------------------
# 3. 반환값(return)이 있는 함수 (결과 돌려받기)
# ---------------------------------------------------------
print("--- [3. return 활용] ---")

def add(num1, num2):
    print(num1, "+", num2, "계산을 시작합니다.")
    return num1 + num2 # 결과를 함수 밖으로 던져줍니다.

# 함수가 돌려준 결과(return값)를 변수에 저장해서 사용할 수 있습니다.
total = add(10, 20)
print("계산 결과:", total)
print()


# ---------------------------------------------------------
# 4. 기본값(Default) 설정하기
# ---------------------------------------------------------
print("--- [4. 기본값 설정] ---")

# age에 값이 들어오지 않으면 기본적으로 20을 사용합니다.
def introduce(name, age=20):
    print("제 이름은", name, "이고, 나이는", age, "세입니다.")

introduce("유리")      # 나이를 안 써도 에러 없이 기본값 20이 출력됨
introduce("맹구", 25)  # 값을 전달하면 기본값을 무시하고 전달된 값을 사용함
print()


# ---------------------------------------------------------
# 5. [심화] 여러 개의 값을 한 번에 반환하기
# ---------------------------------------------------------
print("--- [5. 다중 반환 맛보기] ---")

def calculate(a, b):
    # 파이썬은 콤마(,)를 사용해 여러 값을 한 번에 보낼 수 있습니다.
    return a + b, a - b

plus, minus = calculate(10, 5) # 두 개의 결과값을 각각 변수에 담기
print("합:", plus, "/ 차:", minus)

# 기계(함수)를 한 번만 잘 만들어두면, 재료(매개변수)만 바꿔서 
# 언제든 원하는 만큼 붕어빵(결과)을 찍어낼 수 있습니다!
